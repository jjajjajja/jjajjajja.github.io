# 교환

```c++
#include <bits/stdc++.h>
using namespace std;
int n, k, ans = -1;
bool visited[1000001];

int main(){
    cin >> n >> k;
    int len = to_string(n).size();
    queue<int> q, next_q;
    q.push(n);
    while (k--){
        while (q.size()){
            auto cur = to_string(q.front()); q.pop();
            for (int i = 0 ; i < len ; i++){
                for (int j = i + 1; j < len ; j++){
                    if (i == 0 && cur[j] == '0') continue;
                    swap(cur[i], cur[j]);
                    auto next = stoi(cur);
                    if (!visited[next]){
                        next_q.push(next); visited[next] = true;
                    }
                    swap(cur[j], cur[i]);                   
                }
            }
        }
        if (k == 0){
            for (int l = 1000000; l; l--){
                if (visited[l]) {
                    ans = l;
                    break;
                }
            }
        }
        else{
            q = move(next_q);
            next_q = queue<int>();
            memset(visited, 0, sizeof(visited));
        }
    }
    cout << ans;
}
```
