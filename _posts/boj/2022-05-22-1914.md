# 하노이 탑

```c++
#include <bits/stdc++.h>
using namespace std;
int n;

struct bigInt{
	long long hi;
	long long lo;
	const static long long MOD =  (long long)pow(10, 18);****
	bigInt(long long hi, long long lo) : hi(hi + lo / MOD), lo(lo % MOD){}
	bigInt(long long lo) : hi(lo / MOD), lo(lo % MOD){}
	bigInt operator+(bigInt other){
		return bigInt(this->hi + other.hi, this->lo + other.lo);
	}
	bigInt operator-(bigInt other){
		return bigInt(this->hi - other.hi, this->lo - other.lo);
	}
	string to_string(){
		ostringstream oss;
		if (hi) {
			oss << hi << setw(18) << setfill('0') << lo;
			return oss.str();
		}
		oss << lo;
		return oss.str();
	}
};

void hanoi(int n, int from, int to){
	if (n == 1){
		printf("%d %d\n", from ,to);
		return;
	}
	int via = 6 - from - to;
	hanoi(n - 1, from, via);
	printf("%d %d\n", from, to);
	hanoi(n - 1, via, to);
}

int main(){
	scanf("%d", &n);
	bigInt bi(2);
	for (int i = 1; i < n; i++){
		bi = bi +  bi;
	}
	bi = bi - bigInt(1);
	printf("%s\n", bi.to_string().c_str());
	if (n <= 20) hanoi(n, 1, 3);
}
```
