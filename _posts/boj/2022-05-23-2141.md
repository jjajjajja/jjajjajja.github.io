# 우체국

```c++
#include <bits/stdc++.h>
using namespace std;
using ll = long long;
int n;
vector<pair<ll, ll>> info;
int main(){
	scanf("%d", &n);
	ll s = 0;
	for (int i = 0; i < n; i++){
		ll x, a; scanf("%lld%lld", &x, &a);
		info.push_back({x, a});
		s += a;
	}
	sort(info.begin(), info.end());
	ll cur = 0;
	for (int i = 0 ; i < n; i++){
		cur += info[i].second;
		if (2 * cur >= s){
			printf("%lld", info[i].first);
			break;
		}
	}
}
```
